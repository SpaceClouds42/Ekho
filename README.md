### 1. Add to your project
```kotlin
repositories {
    maven { 
        name = "Modrinth"
        url = uri("https://api.modrinth.com/maven")
    }
}

    // Ekho
    modImplementation("maven.modrinth:ekho:0.3.2")
    include("maven.modrinth:ekho:0.3.2")
}
```
